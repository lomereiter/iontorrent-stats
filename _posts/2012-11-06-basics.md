### Read length distribution


{% highlight r %}
counts <- read.table("_data/counts.dat", header=T)
len <- nrow(counts)
reads.num <- with(counts, reads[1:len-1] - reads[2:len])

xyplot(reads.num ~ I(counts$offset + 1), 
       xlab="Read length, bp",
       ylab="Number of reads",
       pch=19)
{% endhighlight %}
